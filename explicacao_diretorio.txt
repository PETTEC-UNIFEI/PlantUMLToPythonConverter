ESTRUTURA DO PROJETO PLANTUML TO PYTHON CONVERTER
==================================================

PlantUMLToPythonConverter/           # Pasta raiz do projeto
|
|-- app.py                          # Interface desktop (PyWebView)
|-- back_end/                       # Núcleo do conversor PlantUML para Python
|   |-- __init__.py
|   |-- main_cli.py                 # Interface de linha de comando (CLI)
|   |-- plantuml_parser/            # Parser e estruturas de dados do PlantUML
|   |   |-- __init__.py
|   |   |-- parser.py               # Parser sintático do PlantUML
|   |   |-- lexer.py                # Analisador léxico do PlantUML
|   |   |-- data_structures/        # Classes que representam elementos do diagrama
|   |       |-- __init__.py
|   |       |-- plantuml_diagrama.py       # Representa o diagrama inteiro
|   |       |-- plantuml_estrutura_base.py # Classe base para elementos estruturais
|   |       |-- plantuml_classe.py         # Representa classes do diagrama
|   |       |-- plantuml_interface.py      # Representa interfaces
|   |       |-- plantuml_enum.py           # Representa enumerações
|   |       |-- plantuml_atributo.py       # Representa atributos das classes
|   |       |-- plantuml_metodo.py         # Representa métodos das classes
|   |       |-- plantuml_relacionamento.py # Representa relacionamentos entre elementos
|   |       |-- plantuml_pacote.py         # Representa pacotes/namespaces
|   |       |-- plantuml_parametro.py      # Representa parâmetros de métodos
|   |
|   |-- python_generator/           # Geração de código Python
|   |   |-- __init__.py
|   |   |-- main_generator.py       # Orquestrador da geração de código
|   |   |-- utils.py                # Funções utilitárias
|   |   |-- type_mapper.py          # Mapeia tipos PlantUML para Python
|   |   |-- import_manager.py       # Gerencia imports
|   |   |-- structure_generators/   # Geradores para classes, enums, interfaces
|   |       |-- __init__.py
|   |       |-- class_generator.py      # Gera classes Python
|   |       |-- enum_generator.py       # Gera enumerações Python
|   |       |-- interface_generator.py  # Gera interfaces abstratas Python
|
|-- converter.py                    # Script de conversão CLI
|-- front_end/                      # Interface gráfica (HTML, CSS, JS)
|   |-- interface.html              # Estrutura visual da interface
|   |-- style.css                   # Estilos (Tailwind)
|   |-- main.js                     # Lógica de integração frontend
|
|-- data/                           # Dados de entrada e saída
|   |-- diagramas/                  # Diagramas PlantUML de entrada
|   |   |-- exemplo_diagrama.plantuml
|   |   |-- ...                     # Outros diagramas
|   |-- output_generated_code/      # Código Python gerado, um diretório por diagrama
|       |-- <nome_do_diagrama>/     # Estrutura gerada para cada diagrama
|           |-- __init__.py
|           |-- <nome_do_diagrama>.plantuml
|           |-- _diagram_relationships.txt
|           |-- ...                 # Arquivos Python gerados
|
|-- requirements.txt                # Dependências Python
|-- package.json                    # Dependências Node/Tailwind (dev)
|-- .gitignore                      # Arquivos/pastas ignorados pelo git
|-- README.md                       # Documentação principal
|-- explicacao_diretorio.txt        # Este arquivo, explicando a estrutura
|-- tests/                          # Estrutura para testes automatizados
|   |-- back_end_tests/             # Testes do backend
|   |-- front_end_tests/            # Testes do frontend
|   |-- generator/                  # Testes do gerador
|-- .venv/                          # Ambiente virtual Python (não versionado)
|-- node_modules/                   # Dependências Node/Tailwind (não versionado)
|-- __pycache__/                    # Arquivos compilados Python (não versionado)
